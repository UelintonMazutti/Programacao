Módulo 2: Estruturas de Controle

Curso de Lógica de Programação com Linguagem C

1. Estruturas Condicionais (if, else, switch)

As estruturas condicionais permitem que o programa tome decisões com base em condições.

Estrutura if-else:

if (condição) {
    // código executado se a condição for verdadeira
} else {
    // código executado se a condição for falsa
}

Estrutura switch-case:

switch (variável) {
    case valor1:
        // código para valor1
        break;
    case valor2:
        // código para valor2
        break;
    default:
        // código se nenhum caso corresponder
}

2. Operadores Relacionais

Usados para comparar valores em expressões condicionais:

Operador Descrição
== Igual a
!= Diferente de
> Maior que
< Menor que
>= Maior ou igual
<= Menor ou igual

3. Estruturas de Repetição (while, do-while, for)

Estrutura while:

while (condição) {
    // código repetido enquanto a condição for verdadeira
}

Estrutura do-while:

do {
    // código executado pelo menos uma vez
} while (condição);

Estrutura for:

for (inicialização; condição; incremento) {
    // código repetido enquanto a condição for verdadeira
}

4. Comandos break e continue

break: interrompe a execução de um loop ou switch.

continue: pula para a próxima iteração do loop.

for (int i = 0; i < 10; i++) {
    if (i == 5) {
        break; // sai do loop quando i for 5
    }
    if (i % 2 == 0) {
        continue; // pula números pares
    }
    printf("%d ", i);
}

5. Laços Aninhados

Um loop dentro de outro loop:

for (int i = 1; i <= 3; i++) {
    for (int j = 1; j <= 3; j++) {
        printf("%d x %d = %d\n", i, j, i * j);
    }
}

14. Crie um programa que determine se um número é perfeito.

Um número perfeito é igual à soma de seus divisores próprios (excluindo ele mesmo).

#include <stdio.h>

int main() {
    int n, soma = 0;
    printf("Digite um número: ");
    scanf("%d", &n);
    
    for (int i = 1; i < n; i++) {
        if (n % i == 0) {
            soma += i;
        }
    }
    
    printf("%d %s um número perfeito\n", n, soma == n ? "é" : "não é");
    return 0;
}

15. Desenvolva um programa que calcule o MMC entre dois números.

#include <stdio.h>

int main() {
    int a, b, maior, mmc;
    printf("Digite dois números: ");
    scanf("%d %d", &a, &b);
    
    maior = (a > b) ? a : b;
    mmc = maior;
    
    while (1) {
        if (mmc % a == 0 && mmc % b == 0) {
            break;
        }
        mmc += maior;
    }
    
    printf("MMC de %d e %d é %d\n", a, b, mmc);
    return 0;
}

16. Escreva um programa que converta um número decimal para binário.

#include <stdio.h>

int main() {
    int n, bin[32], i = 0;
    printf("Digite um número decimal: ");
    scanf("%d", &n);
    
    while (n > 0) {
        bin[i] = n % 2;
        n /= 2;
        i++;
    }
    
    printf("Binário: ");
    for (int j = i - 1; j >= 0; j--) {
        printf("%d", bin[j]);
    }
    printf("\n");
    
    return 0;
}

18. Crie um programa que resolva uma equação de segundo grau.

#include <stdio.h>
#include <math.h>

int main() {
    float a, b, c, delta, x1, x2;
    printf("Digite os coeficientes a, b e c: ");
    scanf("%f %f %f", &a, &b, &c);
    
    delta = b*b - 4*a*c;
    
    if (delta > 0) {
        x1 = (-b + sqrt(delta)) / (2*a);
        x2 = (-b - sqrt(delta)) / (2*a);
        printf("Duas raízes reais: %.2f e %.2f\n", x1, x2);
    } else if (delta == 0) {
        x1 = -b / (2*a);
        printf("Uma raiz real: %.2f\n", x1);
    } else {
        printf("Não existem raízes reais\n");
    }
    
    return 0;
}

19. Desenvolva um programa que calcule o MDC usando o algoritmo de Euclides.

#include <stdio.h>

int main() {
    int a, b, resto;
    printf("Digite dois números: ");
    scanf("%d %d", &a, &b);
    
    while (b != 0) {
        resto = a % b;
        a = b;
        b = resto;
    }
    
    printf("MDC: %d\n", a);
    return 0;
}
