Módulo 5: Ponteiros e Alocação Dinâmica

Curso de Lógica de Programação com Linguagem C

1. Conceito de Ponteiros

Ponteiros são variáveis que armazenam endereços de memória de outras variáveis.

Declaração e uso básico:

int var = 10;
int *ptr = &var;  // ptr armazena o endereço de var

printf("Valor: %d\n", *ptr);  // Acessa o valor através do ponteiro
printf("Endereço: %p\n", ptr); // Mostra o endereço armazenado

2. Operações com Ponteiros

Principais operações com ponteiros:

Operação Descrição
*ptr Acessa o valor no endereço apontado
&var Obtém o endereço de uma variável
ptr++ Incrementa o ponteiro (avança para próximo elemento)
ptr-- Decrementa o ponteiro (volta para elemento anterior)

3. Ponteiros e Arrays

Arrays e ponteiros têm uma relação estreita em C:

int vetor[5] = {1, 2, 3, 4, 5};
int *ptr = vetor;  // Equivalente a &vetor[0]

// Acesso aos elementos
printf("Primeiro elemento: %d\n", *ptr);
printf("Segundo elemento: %d\n", *(ptr + 1));

4. Alocação Dinâmica de Memória

Funções para gerenciamento de memória dinâmica:

Função Descrição
malloc() Aloca um bloco de memória
calloc() Aloca e inicializa com zero
realloc() Redimensiona um bloco alocado
free() Libera memória alocada

Exemplo:

int *vetor = malloc(5 * sizeof(int));  // Aloca vetor de 5 inteiros
if (vetor != NULL) {
    vetor[0] = 10;  // Atribui valor
    free(vetor);    // Libera memória
}

5. Ponteiros para Funções

Ponteiros podem apontar para funções, permitindo chamá-las dinamicamente:

int soma(int a, int b) { return a + b; }
int (*ptrFunc)(int, int) = soma;  // Ponteiro para função

int resultado = ptrFunc(5, 3);  // Chama a função através do ponteiro

void concatenar(char *dest, const char *src) {
    while (*dest) dest++;  // Encontra o final da string destino
    while (*src) {
        *dest = *src;
        dest++;
        src++;
    }
    *dest = '\0';  // Adiciona terminador nulo
}

11. Desenvolva um programa que realize a multiplicação de matrizes usando ponteiros.

#include <stdio.h>
#include <stdlib.h>

void multiplicarMatrizes(int **a, int **b, int **resultado, int linhasA, int colunasA, int colunasB) {
    for (int i = 0; i < linhasA; i++) {
        for (int j = 0; j < colunasB; j++) {
            resultado[i][j] = 0;
            for (int k = 0; k < colunasA; k++) {
                resultado[i][j] += a[i][k] * b[k][j];
            }
        }
    }
}

int main() {
    int linhasA = 2, colunasA = 3, colunasB = 2;
    
    // Aloca e preenche as matrizes
    int **a = malloc(linhasA * sizeof(int *));
    int **b = malloc(colunasA * sizeof(int *));
    int **resultado = malloc(linhasA * sizeof(int *));
    
    for (int i = 0; i < linhasA; i++) a[i] = malloc(colunasA * sizeof(int));
    for (int i = 0; i < colunasA; i++) b[i] = malloc(colunasB * sizeof(int));
    for (int i = 0; i < linhasA; i++) resultado[i] = malloc(colunasB * sizeof(int));
    
    // Preenche matriz A
    for (int i = 0; i < linhasA; i++)
        for (int j = 0; j < colunasA; j++)
            a[i][j] = i + j;
    
    // Preenche matriz B
    for (int i = 0; i < colunasA; i++)
        for (int j = 0; j < colunasB; j++)
            b[i][j] = i * j;
    
    multiplicarMatrizes(a, b, resultado, linhasA, colunasA, colunasB);
    
    // Imprime resultado
    for (int i = 0; i < linhasA; i++) {
        for (int j = 0; j < colunasB; j++) {
            printf("%d ", resultado[i][j]);
        }
        printf("\n");
    }
    
    // Libera memória
    for (int i = 0; i < linhasA; i++) free(a[i]);
    for (int i = 0; i < colunasA; i++) free(b[i]);
    for (int i = 0; i < linhasA; i++) free(resultado[i]);
    free(a); free(b); free(resultado);
    
    return 0;
}

19. Desenvolva um programa que realize busca binária usando ponteiros.

#include <stdio.h>

int buscaBinaria(int *vetor, int tamanho, int valor) {
    int *inicio = vetor;
    int *fim = vetor + tamanho - 1;
    
    while (inicio <= fim) {
        int *meio = inicio + (fim - inicio) / 2;
        
        if (*meio == valor) return meio - vetor;  // Retorna o índice
        if (*meio < valor) inicio = meio + 1;
        else fim = meio - 1;
    }
    
    return -1;  // Não encontrado
}

int main() {
    int vetor[] = {1, 3, 5, 7, 9, 11, 13};
    int tamanho = sizeof(vetor) / sizeof(vetor[0]);
    int valor = 7;
    
    int posicao = buscaBinaria(vetor, tamanho, valor);
    if (posicao != -1) {
        printf("Valor %d encontrado na posição %d\n", valor, posicao);
    } else {
        printf("Valor não encontrado\n");
    }
    
    return 0;
}
